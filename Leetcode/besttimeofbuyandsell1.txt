class Solution {
public:
    int maxProfit(vector<int> &prices) {
        int len=prices.size();
        if(!len)
            return 0;
        int min=prices[0],res=0;
        for(int i=1;i<len;++i)
        {
            if(prices[i]-min>res)
                res=prices[i]-min;
            if(prices[i]<min)
                min=prices[i];
        }
        return res;
    }
};
